Myco Language Grammar

Grammar Overview

The Myco language follows a context-free grammar defined using Backus-Naur Form (BNF) notation. This grammar defines the syntax rules for valid Myco programs.

Myco's grammar is designed to be simple yet expressive, with clear precedence rules and intuitive syntax.

Token Definitions

Tokens are the basic building blocks of the Myco language:

Keywords

# Control flow
if else for while switch case default

# Function and variable declarations
func let return

# Error handling
try catch

# Block delimiters
end :

Literals

# Numbers
42        # Integer
3.14      # Float

# Strings
"Hello"   # String literal

# Booleans
1         # True
0         # False

Operators

# Arithmetic
+ - * / % **

# Comparison
== != < > <= >=

# Logical
and or not

# Assignment
= += -= *= /=

Expression Grammar

Expressions in Myco follow standard precedence rules:

Primary Expressions

# Primary expressions
primary ::= identifier
           | literal
           | parenthesized_expr
           | function_call
           | array_access
           | object_access

# Parenthesized expressions
parenthesized_expr ::= "(" expression ")"

# Function calls
function_call ::= identifier "(" argument_list ")"

# Argument lists
argument_list ::= expression ("," expression)*
                | Îµ

Operator Precedence

# Precedence levels (highest to lowest)
1. primary (identifiers, literals, parentheses)
2. unary (-, not)
3. multiplicative (*, /, %)
4. additive (+, -)
5. relational (<, >, <=, >=)
6. equality (==, !=)
7. logical_and (and)
8. logical_or (or)
9. assignment (=, +=, etc.)

Statement Grammar

Statements are the top-level constructs in Myco programs:

Variable Declaration

variable_declaration ::= "let" identifier "=" expression ";"

# Examples
let name = "Alice";
let age = 25;
let numbers = [1, 2, 3];

Expression Statement

expression_statement ::= expression ";"

# Examples
print("Hello");
x + y;
counter++;

Block Statement

block ::= statement*

# Examples
{
    let x = 10;
    print(x);
}

Function Grammar

Functions in Myco can be declared with or without type annotations:

Function Declaration

function_declaration ::= "func" identifier "(" parameter_list ")" ["->" type] ":" block "end"

parameter_list ::= parameter ("," parameter)*
                  | Îµ

parameter ::= identifier [":" type]

type ::= "int" | "float" | "string" | "bool"

Function Examples

# Simple function
func greet(name):
    print("Hello,", name);
end

# Function with return type
func add(a: int, b: int) -> int:
    return a + b;
end

Control Flow Grammar

Control flow constructs in Myco:

If Statement

if_statement ::= "if" expression ":" block ["else" block] "end"

# Example
if age >= 18:
    print("Adult");
else:
    print("Minor");
end

For Loop

for_statement ::= "for" identifier "=" expression "to" expression ":" block "end"
                  | "for" identifier "in" expression ":" block "end"

# Examples
for i = 1 to 10:
    print(i);
end

for item in list:
    print(item);
end

Switch Statement

switch_statement ::= "switch" expression ":" case_list ["default" ":" block] "end"

case_list ::= case_item+

case_item ::= "case" expression ":" block

Data Type Grammar

Myco supports several built-in data types:

Array/List

array_literal ::= "[" expression ("," expression)* "]"

# Examples
[1, 2, 3]
["a", "b", "c"]
[x, y, z]

Object/Map

object_literal ::= "{" property_list "}"

property_list ::= property ("," property)*
                  | Îµ

property ::= identifier ":" expression

# Examples
{name: "Alice", age: 25}
{x: 10, y: 20}

Complete BNF Grammar

Here's the complete BNF grammar for the Myco language:

# Program structure
program ::= statement*

# Statements
statement ::= variable_declaration
              | function_declaration
              | if_statement
              | for_statement
              | while_statement
              | switch_statement
              | try_statement
              | return_statement
              | expression_statement
              | block

# Variable declaration
variable_declaration ::= "let" identifier "=" expression ";"

# Function declaration
function_declaration ::= "func" identifier "(" parameter_list ")" ["->" type] ":" block "end"

# Parameters
parameter_list ::= parameter ("," parameter)*
                  | Îµ

parameter ::= identifier [":" type]

# Types
type ::= "int" | "float" | "string" | "bool"

# Control flow
if_statement ::= "if" expression ":" block ["else" block] "end"

for_statement ::= "for" identifier "=" expression "to" expression ":" block "end"
                  | "for" identifier "in" expression ":" block "end"

while_statement ::= "while" expression ":" block "end"

switch_statement ::= "switch" expression ":" case_list ["default" ":" block] "end"

case_list ::= case_item+

case_item ::= "case" expression ":" block

try_statement ::= "try" ":" block "catch" identifier ":" block "end"

return_statement ::= "return" [expression] ";"

# Blocks and statements
block ::= statement*

expression_statement ::= expression ";"

# Expressions
expression ::= assignment

assignment ::= logical_or (assignment_operator assignment)?

assignment_operator ::= "=" | "+=" | "-=" | "*=" | "/="

logical_or ::= logical_and ("or" logical_or)?

logical_and ::= equality ("and" logical_and)?

equality ::= relational (("==" | "!=") relational)?

relational ::= additive (("<" | ">" | "<=" | ">=") additive)?

additive ::= multiplicative (("+" | "-") multiplicative)?

multiplicative ::= unary (("*" | "/" | "%") unary)?

unary ::= ("-" | "not")? primary

# Primary expressions
primary ::= identifier
            | literal
            | parenthesized_expr
            | function_call
            | array_access
            | object_access
            | array_literal
            | object_literal

parenthesized_expr ::= "(" expression ")"

function_call ::= identifier "(" argument_list ")"

argument_list ::= expression ("," expression)*
                | Îµ

array_access ::= primary "[" expression "]"

object_access ::= primary "." identifier

array_literal ::= "[" expression ("," expression)* "]"

object_literal ::= "{" property_list "}"

property_list ::= property ("," property)*
                  | Îµ

property ::= identifier ":" expression

# Literals
literal ::= number | string | boolean

number ::= integer | float

boolean ::= "1" | "0"

# Identifiers
identifier ::= letter (letter | digit | "_")*

letter ::= "a" | "b" | ... | "z" | "A" | "B" | ... | "Z"

digit ::= "0" | "1" | ... | "9"
