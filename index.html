Basic Syntax & Functions 
let name = "Ivy";

# Traditional syntax
func greet(person):
    print("Welcome to the woods,", person, "!");
end

# Arrow syntax for return types
func calculate_power(base: int, exponent: int) -> int:
    return pow(base, exponent);
end

greet(name);
let result = calculate_power(2, 8);  # 256

Control Flow & Error Handling 
# Switch statement
let day = 3;
switch day:
    case 1:
        print("Monday");
    case 2:
        print("Tuesday");
    case 3:
        print("Wednesday");
    default:
        print("Weekend");
end

# Try-catch error handling
try:
    let result = 10 / 0;
catch error:
    print("Caught error:", error);
end
